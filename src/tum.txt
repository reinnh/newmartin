import React, { useState } from "react";
import "./HRPortal.css"; // Assuming you have some styles
import swal from "sweetalert";

const HRPortal = () => {
    const [showOTP, setShowOTP] = useState(false);
    const [username, setUsername] = useState("");
    const [password, setPassword] = useState("");
    const [otp, setOtp] = useState("");
    const [email, setEmail] = useState("");

    // Toggle password visibility
    const togglePassword = () => {
        let input = document.getElementById("password");
        input.type = input.type === "password" ? "text" : "password";
    };

    // Login Function
    const handleLogin = async (e) => {
        e.preventDefault();

        try {
            const response = await fetch("./endpoints/login/", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ username, password }),
            });

            const data = await response.json();

            if (data.success === 0) {
                swal({ text: data.message, icon: "warning", timer: 2000 });
            } else if (data.success === 1) {
                swal({ text: data.message, icon: "success", timer: 2000 }).then(() => {
                    window.location.href = "./?appid=HR";
                });
            } else if (data.success === 2) {
                setShowOTP(true);
                setEmail(data.email);
            }
        } catch (error) {
            swal({ text: "Login failed!", icon: "error", timer: 2000 });
        }
    };

    // OTP Verification
    const handleOTPVerification = async (e) => {
        e.preventDefault();

        try {
            const response = await fetch("./endpoints/login_otp/", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ otp }),
            });

            const data = await response.json();

            if (data.success === 1) {
                swal({ text: data.message, icon: "success", timer: 1000 }).then(() => {
                    window.location.href = "./?appid=HR";
                });
            } else {
                swal({ text: data.message, icon: "warning", timer: 5000 });
            }
        } catch (error) {
            swal({ text: "OTP verification failed!", icon: "error", timer: 5000 });
        }
    };

    return (
        <div className="container">
            <div className="codrops-top">
                <a href="./"><strong>HRMS</strong></a>
                <span className="right">
                    <a href="#"><strong>Help</strong></a>
                </span>
            </div>

            <header>
                <h1><strong>Technical University of Mombasa</strong></h1>
                <h2>Human Resource Management System</h2>
            </header>

            {/* Login Form */}
            {!showOTP ? (
                <section className="main">
                    <form onSubmit={handleLogin}>
                        <h1>
                            <span className="log-in">Log in</span> to <span className="sign-up">TUM</span>
                        </h1>
                        <p className="float">
                            <label htmlFor="login">Username</label>
                            <input
                                type="text"
                                id="login"
                                placeholder="Username or email"
                                value={username}
                                onChange={(e) => setUsername(e.target.value)}
                                autoComplete="off"
                            />
                        </p>
                        <p className="float">
                            <label htmlFor="password">Password</label>
                            <input
                                type="password"
                                id="password"
                                placeholder="Password"
                                value={password}
                                onChange={(e) => setPassword(e.target.value)}
                                autoComplete="off"
                            />
                        </p>
                        <p className="opt">
                            <input type="checkbox" onClick={togglePassword} />
                            <label>Show password</label>
                        </p>
                        <p className="clearfix">
                            <button type="submit">Log in</button>
                            <a href=".?act=ForgotPassword" className="forgotpass">Forgot Password</a>
                        </p>
                    </form>
                </section>
            ) : (
                // OTP Verification
                <section className="main">
                    <form onSubmit={handleOTPVerification}>
                        <h1>A login attempt detected from a new device</h1>
                        <p>Enter the OTP sent to <span className="sign-up">{email}</span></p>
                        <p className="clearfix">
                            <input
                                type="text"
                                id="otp"
                                value={otp}
                                onChange={(e) => setOtp(e.target.value)}
                                autoComplete="off"
                                style={{ fontSize: "18px", width: "300px" }}
                            />
                        </p>
                        <p className="clearfix">
                            <button type="submit" className="btn_otp">Verify OTP</button>
                        </p>
                    </form>
                </section>
            )}
        </div>
    );
};

export default HRPortal;
